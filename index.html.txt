// ── QUALITY SCORE ────────────────────────────
function updateQuality() {
  let score = 0;
  const subject   = document.getElementById('subject').value.trim();
  const scene     = document.getElementById('scene').value.trim();
  const action    = document.getElementById('action').value.trim();
  const artist    = document.getElementById('artist').value.trim();
  const director  = document.getElementById('director').value.trim();
  const negative  = document.getElementById('negative').value.trim();
 
  if (subject)  score += 20;
  if (scene)    score += 12;
  if (action)   score += 8;
  if (artist)   score += 10;
  if (director) score += 8;
  if (negative) score += 5;
 
  const styles    = getActiveChips('styleChips').length;
  const lights    = getActiveChips('lightChips').length;
  const colors    = getActiveChips('colorChips').length;
  const grades    = getActiveChips('gradeChips').length;
  const comps     = getActiveChips('compChips').length;
  const textures  = getActiveChips('textureChips').length;
 
  score += Math.min(styles * 4, 8);
  score += Math.min(lights * 4, 6);
  score += Math.min(colors * 4, 6);
  score += Math.min(grades * 3, 5);
  score += Math.min(comps * 3, 5);
  score += Math.min(textures * 3, 5);
 
  if (document.getElementById('era').value)     score += 3;
  if (document.getElementById('medium').value)  score += 4;
  if (document.getElementById('mood').value)    score += 3;
 
  score = Math.min(score, 100);
 
  qualityFill.style.width = score + '%';
  qualityScore.textContent = score + ' / 100';
 
  // Color shift based on score
  if (score < 30) qualityFill.style.background = 'var(--rust)';
  else if (score < 60) qualityFill.style.background = 'linear-gradient(90deg,var(--gold),var(--gold2))';
  else qualityFill.style.background = 'linear-gradient(90deg,var(--teal),var(--gold),var(--rust))';
}

// ── PROMPT ASSEMBLY ENGINE ───────────────────
 
function buildPrompt(variation) {
  variation = variation || 0;
 
  const subject   = document.getElementById('subject').value.trim();
  const scene     = document.getElementById('scene').value.trim();
  const action    = document.getElementById('action').value.trim();
  const artist    = document.getElementById('artist').value.trim();
  const era       = document.getElementById('era').value;
  const medium    = document.getElementById('medium').value;
  const mood      = document.getElementById('mood').value;
  const timeofday = document.getElementById('timeofday').value;
  const shot      = document.getElementById('shot').value;
  const lens      = document.getElementById('lens').value;
  const aspect    = document.getElementById('aspect').value;
  const quality   = document.getElementById('quality').value;
  const director  = document.getElementById('director').value.trim();
  const negative  = document.getElementById('negative').value.trim();
  const stylize   = parseInt(document.getElementById('stylize').value);
  const coherence = parseInt(document.getElementById('coherence').value);
 
  const styles    = getActiveChips('styleChips');
  const lights    = getActiveChips('lightChips');
  const colors    = getActiveChips('colorChips');
  const grades    = getActiveChips('gradeChips');
  const comps     = getActiveChips('compChips');
  const textures  = getActiveChips('textureChips');
 
  // ── VARIATION MODIFIERS ──
  const variationSeeds = [
    '', 'highly detailed, award-winning', 'cinematic masterpiece', 
    'editorial quality, prize-winning', 'stunning composition',
    'breathtaking detail, gallery quality'
  ];
  const variationMod = variationSeeds[variation % variationSeeds.length];
 
  // ── PLATFORM-SPECIFIC ASSEMBLY ──
  let parts = [];
  let suffix = '';
 
  if (currentPlatform === 'midjourney') {
    // MJ: comma-separated descriptors, params at end
    const mj_s = Math.round(stylize * 10); // 0-1000
    const mj_seed = variation > 0 ? ` --seed ${Math.floor(Math.random()*99999)}` : '';
 
    if (subject)   parts.push(subject);
    if (action)    parts.push(action);
    if (scene)     parts.push(scene);
    if (styles.length)  parts.push(styles.join(', '));
    if (lights.length)  parts.push(lights.join(', '));
    if (timeofday) parts.push(timeofday);
    if (colors.length)  parts.push(colors.join(', '));
    if (grades.length)  parts.push(grades.join(', '));
    if (comps.length)   parts.push(comps.join(', '));
    if (textures.length) parts.push(textures.join(', '));
    if (mood)      parts.push(mood);
    if (era)       parts.push(era);
    if (medium)    parts.push(medium);
    if (shot)      parts.push(shot);
    if (lens)      parts.push(lens);
    if (artist)    parts.push(`in the style of ${artist}`);
    if (director)  parts.push(director);
    if (variationMod) parts.push(variationMod);
 
    suffix = `\n\n${aspect} ${quality} --s ${mj_s}${mj_seed}`;
    if (negative) suffix += `\n--no ${negative}`;
 
  } else if (currentPlatform === 'dalle') {
    // DALL·E 3: full natural language sentences
    let sentences = [];
 
    if (subject || action || scene) {
      let scene_s = '';
      if (subject) scene_s += subject;
      if (action)  scene_s += `, ${action}`;
      if (scene)   scene_s += `. The setting is ${scene}`;
      sentences.push(scene_s + '.');
    }
    if (styles.length || medium) {
      sentences.push(`The image is rendered as ${[...styles, medium].filter(Boolean).join(', ')}.`);
    }
    if (lights.length || timeofday) {
      sentences.push(`Lighting: ${[...lights, timeofday].filter(Boolean).join(', ')}.`);
    }
    if (colors.length || grades.length) {
      sentences.push(`Color palette: ${[...colors, ...grades].filter(Boolean).join(', ')}.`);
    }
    if (mood)  sentences.push(`The mood is ${mood}.`);
    if (era)   sentences.push(`${era} aesthetic.`);
    if (shot)  sentences.push(`Shot as a ${shot}.`);
    if (lens)  sentences.push(`${lens}.`);
    if (comps.length)    sentences.push(`Composition: ${comps.join(', ')}.`);
    if (textures.length) sentences.push(`Surface quality: ${textures.join(', ')}.`);
    if (artist)  sentences.push(`Inspired by the visual style of ${artist}.`);
    if (director) sentences.push(director);
    if (variationMod) sentences.push(variationMod + '.');
 
    parts = sentences;
 
  } else if (currentPlatform === 'stable') {
    // SD: weighted prompts, structured with emphasis
    if (subject)   parts.push(`(${subject}:1.4)`);
    if (action)    parts.push(`(${action}:1.2)`);
    if (scene)     parts.push(scene);
    if (styles.length)   parts.push(styles.map(s=>`(${s}:1.3)`).join(', '));
    if (lights.length)   parts.push(lights.map(l=>`(${l}:1.2)`).join(', '));
    if (timeofday) parts.push(timeofday);
    if (colors.length)   parts.push(colors.join(', '));
    if (grades.length)   parts.push(grades.join(', '));
    if (comps.length)    parts.push(comps.join(', '));
    if (textures.length) parts.push(textures.map(t=>`(${t}:1.1)`).join(', '));
    if (mood)      parts.push(`(${mood}:1.2)`);
    if (era)       parts.push(era);
    if (medium)    parts.push(`(${medium}:1.3)`);
    if (shot)      parts.push(shot);
    if (lens)      parts.push(lens);
    if (artist)    parts.push(`(by ${artist}:1.35)`);
    if (director)  parts.push(director);
    if (variationMod) parts.push(variationMod);
 
    if (negative) suffix = `\n\nNEGATIVE PROMPT:\n${negative}`;
 
  } else {
    // Generic (Firefly / Ideogram / other)
    if (subject)   parts.push(subject);
    if (action)    parts.push(action);
    if (scene)     parts.push(scene);
    if (styles.length)   parts.push(styles.join(', '));
    if (lights.length)   parts.push(lights.join(', '));
    if (timeofday) parts.push(timeofday);
    if (colors.length)   parts.push(colors.join(', '));
    if (grades.length)   parts.push(grades.join(', '));
    if (comps.length)    parts.push(comps.join(', '));
    if (textures.length) parts.push(textures.join(', '));
    if (mood)      parts.push(mood);
    if (era)       parts.push(era);
    if (medium)    parts.push(medium);
    if (shot)      parts.push(shot);
    if (lens)      parts.push(lens);
    if (artist)    parts.push(`style inspired by ${artist}`);
    if (director)  parts.push(director);
    if (variationMod) parts.push(variationMod);
    if (negative) suffix = `\n\nAVOID: ${negative}`;
  }
 
  // Filter empties
  parts = parts.filter(Boolean);
 
  if (parts.length === 0) return null;
 
  const separator = currentPlatform === 'dalle' ? '\n' : ', ';
  const prompt = parts.join(separator) + suffix;
 
  return prompt;
}
 
// ── TYPEWRITER EFFECT ────────────────────────
function typewriterSet(el, text, speed) {
  speed = speed || 8;
  // Remove old cursor BEFORE wiping content
  const oldCursor = el.querySelector('.cursor');
  if (oldCursor) oldCursor.remove();
 
  el.classList.remove('empty');
  el.textContent = '';
 
  let i = 0;
  let cursor = document.createElement('span');
  cursor.className = 'cursor';
  el.appendChild(cursor);
 
  const interval = setInterval(() => {
    if (i < text.length) {
      el.insertBefore(document.createTextNode(text[i]), cursor);
      i++;
      el.scrollTop = el.scrollHeight;
    } else {
      clearInterval(interval);
      generating = false;
      btnGenerate.classList.remove('loading');
    }
  }, speed);
}
 
// ── GENERATE ─────────────────────────────────
function generate(variation) {
  if (generating) return;
  variation = variation || 0;
 
  const prompt = buildPrompt(variation);
 
  if (!prompt) {
    showToast('⚠ Add a subject or scene description first');
    return;
  }
 
  generating = true;
  btnGenerate.classList.add('loading');
 
  // Update output
  updatePlatformNote();
  typewriterSet(promptOutput, prompt, variation > 0 ? 5 : 8);
 
  // Update metrics
  const words = prompt.split(/\s+/).filter(Boolean).length;
  const chars = prompt.length;
  const score = parseInt(qualityScore.textContent) || 0;
 
  mWords.textContent = words;
  mChars.textContent = chars;
  mScore.textContent = score > 0 ? score : '—';
 
  // Save to history
  addHistory(prompt, score);
}
 
// ── HISTORY ──────────────────────────────────
function addHistory(prompt, score) {
  const platform = currentPlatform;
  const item = { prompt, score, platform, time: new Date() };
  promptHistory.unshift(item);
  if (promptHistory.length > 5) promptHistory.pop();
  renderHistory();
}
 
function renderHistory() {
  if (promptHistory.length === 0) return;
  historyList.innerHTML = '';
  promptHistory.forEach((item, idx) => {
    const div = document.createElement('div');
    div.className = 'history-item';
    div.innerHTML = `
      <div class="history-item-text">${escHtml(item.prompt)}</div>
      <div class="history-item-meta">
        <span>${timeAgo(item.time)}</span>
        <span class="history-badge">${item.platform.toUpperCase()} · Q:${item.score}</span>
      </div>`;
    div.addEventListener('click', () => {
      promptOutput.classList.remove('empty');
      promptOutput.textContent = item.prompt;
      const cursor = document.createElement('span');
      cursor.className = 'cursor';
      promptOutput.appendChild(cursor);
      showToast('✦ Restored from history');
    });
    historyList.appendChild(div);
  });
}
 
function timeAgo(date) {
  const s = Math.floor((new Date() - date) / 1000);
  if (s < 10) return 'just now';
  if (s < 60) return `${s}s ago`;
  return `${Math.floor(s/60)}m ago`;
}
function escHtml(t) {
  return t.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
}
 
// ── COPY ──────────────────────────────────────
function copyPrompt() {
  // Extract text content excluding the cursor span
  let text = '';
  promptOutput.childNodes.forEach(node => {
    if (node.nodeType === Node.TEXT_NODE) text += node.textContent;
  });
  text = text.trim();
  if (!text || text.startsWith('//')) {
    showToast('⚠ No prompt to copy yet');
    return;
  }
  navigator.clipboard.writeText(text).then(() => {
    copyBtn.textContent = 'COPIED ✓';
    copyBtn.classList.add('copied');
    showToast('✦ Prompt copied to clipboard');
    setTimeout(() => {
      copyBtn.textContent = 'COPY';
      copyBtn.classList.remove('copied');
    }, 2200);
  }).catch(() => {
    // Fallback
    const ta = document.createElement('textarea');
    ta.value = text;
    ta.style.position = 'fixed'; ta.style.opacity = '0';
    document.body.appendChild(ta); ta.select();
    try { document.execCommand('copy'); showToast('✦ Copied!'); } catch(e) {}
    document.body.removeChild(ta);
  });
}
 
// ── TOAST ────────────────────────────────────
let toastTimer = null;
function showToast(msg) {
  toast.textContent = msg;
  toast.classList.add('show');
  if (toastTimer) clearTimeout(toastTimer);
  toastTimer = setTimeout(() => toast.classList.remove('show'), 2500);
}
